#!/bin/bash

# Спрашиваем у пользователя имя каталога
echo "Введите имя каталога:"
read dir_name

# Вариант 1: проверяем, что каталог существует и не является символической ссылкой
if [[ -d "$dir_name" && ! -L "$dir_name" ]]; then
  # Находим самый большой файл в указанном каталоге
  largest_file=$(find "$dir_name" -maxdepth 1 -type f -not -name "." -printf "%s %p\n" | sort -nr | head -n 1 | cut -d " " -f 2)

  # Записываем информацию о выполненном действии в файл /var/log/фамилия.log
  echo "$(date) - Найден самый большой файл в каталоге $dir_name: $largest_file" >> /var/log/ivanov.log
else
  # Если каталог не найден или является символической ссылкой, выводим сообщение об ошибке
  echo "Ошибка: каталог $dir_name не найден или является символической ссылкой." >&2
  # Записываем информацию об ошибке в файл /var/log/фамилия.log
  echo "$(date) - Ошибка: каталог $dir_name не найден или является символической ссылкой." >> /var/log/ivanov.log
fi

# Вариант 2: спрашиваем у пользователя, хочет ли он продолжить работу с указанным каталогом
echo "Вы хотите продолжить работу с каталогом $dir_name? (y/n)"
read answer

if [[ "$answer" == "y" ]]; then
  # Находим самый большой файл в указанном каталоге
  largest_file=$(find "$dir_name" -maxdepth 1 -type f -not -name "." -printf "%s %p\n" | sort -nr | head -n 1 | cut -d " " -f 2)

  # Записываем информацию о выполненном действии в файл /var/log/фамилия.log
  echo "$(date) - Найден самый большой файл в каталоге $dir_name: $largest_file" >> /var/log/ivanov.log
else
  # Если пользователь не хочет продолжать работу с указанным каталогом, выводим сообщение об ошибке
  echo "Работа с каталогом $dir_name прервана пользователем." >&2
  # Записываем информацию об ошибке в файл /var/log/фамилия.log
  echo "$(date) - Работа с каталогом $dir_name прервана пользователем." >> /var/log/ivanov.log
fi
